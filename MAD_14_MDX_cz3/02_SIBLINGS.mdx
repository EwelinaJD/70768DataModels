
--------------------------------------------------------------------------------------------------------------------------------
--	Siblings
--------------------------------------------------------------------------------------------------------------------------------

--	Returns the siblings of a specified member, including the member itself.
--------------------------------------------------------------------------------------------------------------------------------

	SELECT 
		[Measures].[Internet Sales Amount]										ON 0,
		[Product].[Product Categories].[Subcategory].[Socks].PARENT.CHILDREN	ON 1
	FROM [Adventure Works]		
	;

	SELECT 
		[Measures].[Internet Sales Amount]								ON 0,
		[Product].[Product Categories].[Subcategory].[Socks].SIBLINGS	ON 1
	FROM [Adventure Works]		
	;
